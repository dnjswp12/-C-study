// const 변수는 항상 선언과 동시에 초기화를 진행.
// 상수 포인터는 위치에 따라 변경 할 수 없는 값이 다르다.

void P_ConstVar
{
	int a = 1; 
	int b = 2;

	const int *p1 = &a;
	int* const p2 = &b;

	p1 = &b;
	*p1 = 3; // 오류 발생
/* 포인터 변수가 가리키는 주소가 가지고 있는 값을 변경하지 못함. 대상체만 변경할 수 있으며, 대상체의 값을 변경하지 못한다. 이 경우는 아래와 같다. 
	 const int a = 1;
	 const int *p = &a;
	 a = 2;
	 *p = 2;
// 일반 변수도 const가 붙으면 값을 변경할 수 없는 것    과 같이 "const 포인터 변수"도 마찬가지이다.  
*/

	p2 = &b; //  오류발생
	*p2 = 3; 
/* 포인터 변수가 가리키는 주소의 값을 변경할 수 있다.
포인터 변수가 가리키는 주소값을 변경할 수 없다. 
즉, 대상체를 변경 할 수 없으며 대상체의 값만 변경할 수 있다.
 */
}
